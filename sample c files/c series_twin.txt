#include <stdio.h>

void main()
{
    int n;
    printf("enter n");
    scanf("%d",&n);
    int s,l;
    int m=1;
    for(int i=1;i<=n;i++)
    {
        l=s-i+2;
        for(int j=1;j<i;j++)
        {
            printf("%d\t",l++);
        }
        
        for(int k=1;k<((n-i)+2);k++)
        {
            printf("%d\t",(m++));
        }
        s=(1+i)*n;
        m=1+(i*n);
        printf("\n");
    }
}
